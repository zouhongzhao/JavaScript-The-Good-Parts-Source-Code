Function.prototype.method = function (name, func) {
    this.prototype[name] = func;
    return this;
};

// Add a method conditionally.

Function.prototype.method = function (name, func) {
    if (!this.prototype[name]) {
        this.prototype[name] = func;
        return this;
    }
};
